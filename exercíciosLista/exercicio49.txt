Algoritmo "parImpar"

Var
   numero, soma, cont, pares, impar: inteiro
Inicio
   soma <- 0
   cont <- 1
   pares <- 0
   impar <- 0
   Enquanto (cont <= 6) faca
      Escreval("Digite um número inteiro:")
      Leia(numero)
      Se(numero % 2 = 0)entao
         pares <- pares + 1
      Senao
         impar <- impar + 1
      FimSe
      cont <- cont + 1
      soma <- soma + numero
   FimEnquanto
   Escreval("A soma dos número digitados é: ", soma)
   Escreval(pares, " números são pares")
   Escreval(impar, " números são ímpares")
Fimalgoritmo
