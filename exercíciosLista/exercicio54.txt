      Leia(altura)

      Se(peso > 90)entao
         maisNoventa <- maisNoventa + 1
      FimSe

         menosCinquenta <- menosCinquenta + 1
      FimSe
      Se(peso > 100) e (altura > 1.90)entao
         maisCem <- maisCem + 1
      FimSe
      cont <- cont + 1
   FimEnquanto
