Algoritmo "funcaoMaior"

funcao maior(a, b, c: real): real
Var
   m: real
Inicio
   Se((a > b) e (a > c))entao
      m <- a
   Senao
      Se((b > a) e (b > c))entao
         m <- b
      Senao
         Se((c > a) e (c > b))entao
            m <- c
         FimSe
      FimSe
   FimSe
   Retorne m
FimFuncao
