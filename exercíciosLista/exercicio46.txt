Algoritmo "soma6ao100"

Var
   cont, soma: inteiro
Inicio
   soma <- 0
   cont <- 6
   Enquanto (cont <= 100) faca
      Se(cont < 100)entao
         Escreva(cont, "+")
         Soma <- soma + cont
      Senao
         Escreva(cont)
         Soma <- soma + cont
      FimSe
      cont <- cont + 2
   FimEnquanto
   Escreva("=", soma)
Fimalgoritmo
