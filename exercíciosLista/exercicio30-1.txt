Algoritmo "triangulo"
Var
   lado1, lado2, lado3: real
   equilatero, escaleno, tri, isosceles, trianguloRet: logico
Inicio
   Escreva("Digite o primeiro lado do triângulo:")
   Leia(lado1)
   Escreva("Digite o segundo lado do triângulo:")
   Leia(lado2)
   Escreva("Digite o terceiro lado do triângulo:")
   Leia(lado3)
   tri <- (lado1 < lado2 + lado3) e (lado2 < lado1 + lado3) e (lado3 < lado1 + lado3)
   equilatero <- (lado1 = lado2) e (lado2 = lado3)
   escaleno <- (lado1 <> lado2) e (lado2 <> lado3) e (lado3 <> lado1)
   Escreval("Forma um triângulo? ", tri)
   Se (equilatero = falso) e (escaleno = falso) e (tri = verdadeiro) entao
      trianguloRet <- (lado1 * lado1 = lado2 * lado2 + lado3 * lado3) ou (lado2 * lado2 = lado1 * lado1 + lado3 * lado3) ou (lado3 * lado3 = lado2 * lado2 + lado1 * lado1)
      isosceles <- verdadeiro
      Escreval("O triângulo é isósceles? ", isosceles)
      Escreval("O triângulo é equilátero? ", equilatero)
      Escreval("O triângulo é escaleno? ", escaleno)
      Escreval("O triângulo é retângulo? ", trianguloRet)
   Senao
      Se (tri = verdadeiro) entao
         isosceles <- falso
         trianguloRet <- (lado1 * lado1 = lado2 * lado2 + lado3 * lado3) ou (lado2 * lado2 = lado1 * lado1 + lado3 * lado3) ou (lado3 * lado3 = lado2 * lado2 + lado1 * lado1)
         Escreval("O triângulo é isósceles? ", isosceles)
         Escreval("O triângulo é equilátero? ", equilatero)
         Escreval("O triângulo é escaleno? ", escaleno)
         Escreval("O triângulo é retângulo? ", trianguloRet)
      FimSe
   FimSe

Fimalgoritmo
