Algoritmo "ordenaVetor"

Var
   vet: vetor[1..4] de inteiro
   i, j, troca: inteiro
Inicio
   Para i <- 1 ate 4 faca
      Escreva("Digite um valor: ")
      Leia(vet[i])
   FimPara
   Para i <- 1 ate 3 faca
      Para j <- i + 1 ate 4 faca
         Se(vet[i] > vet[j])entao
            troca <- vet[i]
            vet[i] <- vet[j]
            vet[j] <- troca
         FimSe
      FimPara
   FimPara
   Para i <- 1 ate 4 faca
      Escreva("{", vet[i], "}")
   FimPara
Fimalgoritmo
