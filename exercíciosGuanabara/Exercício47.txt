Algoritmo "quantosEntre0e10"

Var
   cont, valor, contNumeros: inteiro
Inicio
   contNumeros <- 0
   Para cont <- 1 ate 6 faca
      Escreval("Digite um valor")
      Leia(valor)
      Se(valor >= 0) e (valor <= 10)entao
         contNumeros <- contNumeros + 1
      FimSe
   FimPara
   Escreval("O número de valores digitados que são menor que 10 e maior que 0 são ", contNumeros)
Fimalgoritmo
