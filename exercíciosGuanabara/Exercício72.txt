Algoritmo "lerMatriz"

Var
   matri: vetor[1..3, 1..2] de inteiro
   i, j: inteiro

Inicio
   Para i <- 1 ate 3 faca
      Para j <- 1 ate 2 faca
         Escreva("Digite o valor da posição [", i, ",", j, "]:")
         Leia(matri[i,j])
      FimPara
   FimPara
   Para i <- 1 ate 3 faca
      Para j <- 1 ate 2 faca
         Escreva(matri[i, j])
      FimPara
      Escreval()
   FimPara
FimAlgoritmo
