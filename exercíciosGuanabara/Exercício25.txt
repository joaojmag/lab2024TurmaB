   Leia(nota2)
   media <- (nota1 + nota2) / 2
   Escreval("MÃ©dia: ", media)
   Se (media >= 9) entao

   Senao

        Senao

             Senao

                     Escreval("Aproveitamento: D")

                       Senao
                            Escreval("Aproveitamento: F")

                  FimSe
             FimSe
        FimSe
