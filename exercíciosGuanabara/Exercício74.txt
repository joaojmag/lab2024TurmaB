Algoritmo "matrizIdentidade"

Var
   numeros: vetor[1..3, 1..3] de inteiro
   i, j: inteiro
Inicio
   Para i <- 1 ate 3 faca
      Para j<- 1 ate 3 faca
         Se(j = i)entao
            numeros[j, i] <- 1
         Senao
            numeros[j, i] <- 0
         FimSe
      FimPara
   FimPara
   Para i <- 1 ate 3 faca
      Para j <- 1 ate 3 faca
         Escreva(numeros[i, j])
      FimPara
      Escreval()
   FimPara
FimAlgoritmo
