Algoritmo "fatorialComResposta"
Var

   numero, cont, fatorial: inteiro
   resp: caractere
Inicio
   Repita
      Escreval("Digite um número:")
      Leia(numero)
      cont <- numero
      fatorial <- 1
      Repita
         fatorial <- fatorial * cont
         cont <- cont - 1
      Ate(cont < 1)
      Escreval("O fatorial de ", numero, " é igual a ", fatorial)
      Escreval("Quer continuar?")
      Leia(resp)
   Ate(resp = "N")
Fimalgoritmo
